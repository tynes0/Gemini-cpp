Gemini-cpp is a lightweight C++ client library for the Google Gemini API. It provides a simple, modern interface for sending prompts, handling responses, and working with structured model outputs. Designed with minimal dependencies and straightforward usage in mind, it lets you integrate Gemini into C++ projects without dealing with HTTP boilerplate or complex setup.

75 {
76 using Traits = function_traits<std::decay_t<Func>>;
77
78 FunctionDeclaration decl;
79 decl.name = name;
80 decl.description = description;
81 decl.parameters = generateSchema<typename Traits::ArgsTuple>(argNames);

83 auto invoker = [func = std::forward<Func>(func), argNames](const nlohmann::json& args) -> nlohmann::json
84 {
85 try
86 {
87 auto tupleArgs = Internal::jsonToTuple<typename Traits::ArgsTuple>(args, argNames);
88
89 if constexpr (std::is_void_v<typename Traits::ReturnType>)
90 {
91 std::apply(func, tupleArgs);
92 return {{"result", "ok"}};
93 } else
94 {
95 return std::apply(func, tupleArgs);
96 }
97 }
98 catch (const std::exception& e)

101 return {{"error", e.what()}};
102 }
103 };
